public class Solution {
  public int FirstUniqChar(string s) {
    Dictionary < char, int > occurrences = new Dictionary < char, int > ();

    int firstNumber = 0;
    foreach(char c in s) {
      occurrences.TryAdd(c, 0);
      occurrences[c]++;
    }

    for (int i = 0; i < s.Length; i++) {
      char c = s[i];
      if (occurrences[c] == 1) {
        return i;
      }
    }

    return -1;
  }
}
