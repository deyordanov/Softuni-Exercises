public class Solution {
  public int MinCostClimbingStairs(int[] cost) {
    int[] newArray = new int[cost.Length + 1];
    Array.Copy(cost, newArray, cost.Length);

    for (int i = newArray.Length - 3; i >= 0; i--) {
      newArray[i] += Math.Min(
        newArray[i + 1],
        newArray[i + 2]);
    }

    return Math.Min(newArray[0], newArray[1]);
  }
}
