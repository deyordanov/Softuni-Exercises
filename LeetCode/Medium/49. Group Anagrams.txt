public class Solution {
  public IList < IList < string >> GroupAnagrams(string[] strs) {
    Dictionary < string, IList < string >> elements = new Dictionary < string, IList < string >> ();

    foreach(string str in strs) {
      string result = CalculateStringValue(str);
      elements.TryAdd(result, new List < string > ());
      elements[result].Add(str);
    }

    return elements.Values.ToList();
  }

  string CalculateStringValue(string str) {
    char[] result = str.ToCharArray();
    Array.Sort(result);
    return new string(result);
  }
}
