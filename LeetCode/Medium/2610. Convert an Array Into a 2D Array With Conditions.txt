public class Solution {
    public IList<IList<int>> FindMatrix(int[] nums) {
        Dictionary<int, int> numbers = new Dictionary<int, int>();
        int rows = 0;
        int rowSize = 0;
        for(int i = 0; i < nums.Length; i++)
        {
            if(!numbers.ContainsKey(nums[i]))
            {
                numbers.Add(nums[i], 0);
                rowSize++;
            }

            numbers[nums[i]]++;
            if(numbers[nums[i]] > rows)
            {
                rows++;
            }
        }   

        int[][] matrix = new int[rows][];

        int currentRow = 0;
        int currentCol = 0;
        while(currentRow < rows)
        {
            matrix[currentRow] = new int[rowSize];
            foreach(int number in numbers.Keys)
            {
                if(numbers[number] > 0)
                {
                    matrix[currentRow][currentCol] = number;
                    numbers[number]--;
                    currentCol++;

                    if(numbers[number] == 0)
                    {
                        rowSize--;
                        numbers.Remove(number);
                    }
                }
            }

            currentRow++;
            currentCol = 0;
        }

        return matrix;
    }
}
