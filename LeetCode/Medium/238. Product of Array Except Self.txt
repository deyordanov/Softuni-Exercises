public class Solution {
  public int[] ProductExceptSelf(int[] nums) {
    int[] result = new int[nums.Length];
    Array.Fill(result, 1);

    int leftProduct = 1;
    for (int i = 0; i < nums.Length; i++) {
      result[i] *= leftProduct;

      leftProduct *= nums[i];
    }

    int rightProduct = 1;
    for (int i = nums.Length - 1; i >= 0; i--) {
      result[i] *= rightProduct;

      rightProduct *= nums[i];
    }

    return result;
  }
}
